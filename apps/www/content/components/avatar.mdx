## Usage

```tsx
import { Avatar, AvatarImage, AvatarFallback } from "glasswave";
```

```tsx
<Avatar>
  <AvatarImage src="https://avatars.githubusercontent.com/u/66132191" />
  <AvatarFallback>DI</AvatarFallback>
</Avatar>
```
