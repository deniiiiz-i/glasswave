## Usage

```tsx
import {
  InputOTP,
  InputOTPGroup,
  InputOTPSlot,
  InputOTPSeparator,
} from "glasswave";

<InputOTP maxLength={6}>
  <InputOTPGroup>
    <InputOTPSlot index={0} />
    …
  </InputOTPGroup>
  <InputOTPSeparator />
  <InputOTPGroup>…</InputOTPGroup>
</InputOTP>
```
