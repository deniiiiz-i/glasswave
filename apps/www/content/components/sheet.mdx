## Usage

```tsx
import {
  Sheet,
  SheetTrigger,
  SheetContent,
  SheetHeader,
  SheetTitle,
  SheetDescription,
} from "glasswave";

<Sheet>
  <SheetTrigger asChild>
    <Button>Open</Button>
  </SheetTrigger>
  <SheetContent side="right">…</SheetContent>
</Sheet>
```

## API Reference

Built on [Radix UI Dialog](https://www.radix-ui.com/primitives/docs/components/dialog).

### SheetContent

| Prop   | Type                                     | Default   |
| ------ | ---------------------------------------- | --------- |
| `side` | `"top" \| "right" \| "bottom" \| "left"` | `"right"` |
